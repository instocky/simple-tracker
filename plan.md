### –ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏–π virtual environment (–†–ï–ö–û–ú–ï–ù–î–£–ï–ú–´–ô)

```python
import os
import sys
import subprocess
import venv

def ensure_virtual_environment():
    """–ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ —Å–æ–∑–¥–∞–µ—Ç –∏ –∞–∫—Ç–∏–≤–∏—Ä—É–µ—Ç virtual environment"""

    project_root = os.path.dirname(os.path.abspath(__file__))
    venv_path = os.path.join(project_root, 'venv')

    if not os.path.exists(venv_path):
        print("–°–æ–∑–¥–∞–Ω–∏–µ –≤–∏—Ä—Ç—É–∞–ª—å–Ω–æ–≥–æ –æ–∫—Ä—É–∂–µ–Ω–∏—è...")
        venv.create(venv_path, with_pip=True)

    # –ü–æ–ª—É—á–∞–µ–º –ø—É—Ç—å –∫ Python –≤ venv
    if sys.platform == 'win32':
        python_path = os.path.join(venv_path, 'Scripts', 'python.exe')
        pip_path = os.path.join(venv_path, 'Scripts', 'pip.exe')
    else:
        python_path = os.path.join(venv_path, 'bin', 'python')
        pip_path = os.path.join(venv_path, 'bin', 'pip')

    # –ü—Ä–æ–≤–µ—Ä—è–µ–º –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏
    try:
        result = subprocess.run([python_path, '-c', 'import flask'],
                              capture_output=True, text=True)
        if result.returncode != 0:
            print("–£—Å—Ç–∞–Ω–æ–≤–∫–∞ –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–µ–π –≤ –≤–∏—Ä—Ç—É–∞–ª—å–Ω–æ–µ –æ–∫—Ä—É–∂–µ–Ω–∏–µ...")
            packages = ['flask', 'flask-cors']
            subprocess.run([pip_path, 'install'] + packages, check=True)
    except Exception as e:
        print(f"–û—à–∏–±–∫–∞ —Å–æ–∑–¥–∞–Ω–∏—è –≤–∏—Ä—Ç—É–∞–ª—å–Ω–æ–≥–æ –æ–∫—Ä—É–∂–µ–Ω–∏—è: {e}")
        sys.exit(1)

    return python_path, venv_path
```
